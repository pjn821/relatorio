#include <iostream>
using namespace std;

int main() 
{
  string sA = "The C++at home";
  string sB = "The C ++ at home"; //The-C-plus-plus-at-home
  string new_string = "";
  for (int i = 0; i < sB.length(); i++){
    char current_character = sB[i];
    char next_character_is_alnum = isalnum(sB[i+1]);
    if (isalnum(current_character)) { 
      new_string += current_character;
      
    } else if (current_character == '+'){
      // "C++at" 
      // "C-plus-plusat"
      new_string += "-plus";
      if (next_character_is_alnum){
        new_string += "-";
      // "C-plus-plus-at"
      }
    } else {
      new_string += '-';
    }
  }
  cout << sB << endl;
  cout << new_string << endl;
}
